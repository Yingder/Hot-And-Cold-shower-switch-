<script>

(()=> {

  const el = id => document.getElementById(id);

  const hot = el('hot'), cold = el('cold'), loops = el('loops'), warmup = el('warmup');

  const phase = el('phase'), clock = el('clock'), loopInfo = el('loopInfo');

  const btnStart = el('btnStart'), btnPause = el('btnPause'), btnReset = el('btnReset');

 

  let plan = [], idx = 0, remaining = 0, timer=null, paused=false, currentLoop=0, totalLoops=0;

 

  function speak(text){

    try{

      const u = new SpeechSynthesisUtterance(text);

      u.lang = 'zh-CN'; u.rate = 1.05; u.pitch = 1;

      speechSynthesis.cancel(); speechSynthesis.speak(u);

    }catch(e){}

  }

  function buzz(ms=200){ if(navigator.vibrate) navigator.vibrate(ms); }

  function fmt(s){ const m=Math.floor(s/60).toString().padStart(2,'0'); const ss=(s%60).toString().padStart(2,'0'); return `${m}:${ss}`; }

 

  function buildPlan(){

    plan = [];

    totalLoops = Number(loops.value)||6;

    const h = Number(hot.value)||30, c = Number(cold.value)||30, w = Number(warmup.value)||0;

    if(w>0) plan.push({type:'warm', secs:w, label:'热身'});

    for(let i=1;i<=totalLoops;i++){

      plan.push({type:'hot', secs:h, label:`热水 第${i}轮`});

      plan.push({type:'cold', secs:c, label:`冷水 第${i}轮`});

    }

    plan.push({type:'done', secs:0, label:'完成'});

  }

 

  function show(){

    const step = plan[idx]||{type:'idle', label:'待开始', secs:0};

    phase.innerHTML = step.type==='hot' ? `<span class="hot">热水</span>` :

                      step.type==='cold' ? `<span class="cold">冷水</span>` :

                      step.type==='warm' ? `热身` :

                      step.type==='done' ? `完成` : '待开始';

    clock.textContent = fmt(remaining);

    loopInfo.textContent = `${currentLoop} / ${totalLoops}`;

  }

 

  function cue(step, enter=true){

    if(step.type==='hot' && enter){ speak(`开始热水 ${step.secs} 秒`); buzz(150); }

    if(step.type==='cold' && enter){ speak(`切换冷水 ${step.secs} 秒`); buzz([80,80,80]); }

    if(step.type==='warm' && enter){ speak(`热身 ${step.secs} 秒`); buzz(120); }

    if(step.type==='done'){ speak('完成，温水收尾'); buzz([200,60,200]); }

  }

 

  function tick(){

    if(paused) return;

    if(remaining>0){ remaining--; show(); return; }

    idx++;

    const step = plan[idx];

    if(!step){ stopAll(); return; }

    if(step.type==='hot') currentLoop++;

    remaining = step.secs;

    cue(step,true); show();

    if(step.type==='done'){ stopAll(); }

  }

 

  function start(){

    if(timer) return;

    buildPlan(); idx=0; currentLoop=0; remaining=plan[0]?.secs||0;

    cue(plan[0], true); show();

    paused=false;

    timer = setInterval(tick, 1000);

    btnStart.disabled = true; btnPause.disabled = false; btnReset.disabled = false;

  }

 

  function pause(){

    if(!timer) return;

    paused = !paused;

    btnPause.textContent = paused ? '继续' : '暂停';

    if(paused){ speak('暂停'); } else { speak('继续'); }

  }

 

  function stopAll(){

    clearInterval(timer); timer=null; paused=false;

    btnStart.disabled = false; btnPause.disabled = true; btnReset.disabled = true;

  }

 

  function reset(){

    stopAll();

    phase.textContent = '待开始'; clock.textContent='00:00'; loopInfo.textContent = '0 / 0';

    speak('已重置');

  }

 

  btnStart.addEventListener('click', start);

  btnPause.addEventListener('click', pause);

  btnReset.addEventListener('click', reset);

 

  // 解锁语音引擎（需要一次用户点击）

  document.body.addEventListener('click', ()=>{ try{ speechSynthesis.getVoices(); }catch(e){} }, {once:true});

})();

</script>
